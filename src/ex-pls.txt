function formatName(user) {
  return user.firstName + " " + user.lastName;
}

function getGreeting(user) {
  if (user.firstName === "Камиль") {
    return <h1>Здравствуй, Камиль!</h1>;
  } else if (user.lastName === "Варин") {
    return <h1>Здравствуй, Варин.</h1>;
  }
}

const user = {
  firstName: "Камиль",
  lastName: "Моревна",
  age: 25,
};

const user2 = {
  firstName: "Ильгиз",
  lastName: "Варин",
  age: 25,
};

const user3 = {
  name:'Куджа',
  firstName: "Ильгиз",
  lastName: "Варин",
  age: 25,
};

const element1 = getGreeting(user);
const element2 = getGreeting(user2);

function Welcome(props) {
  return <h1>Привет, {props.name}</h1>;
}

const element3 = Welcome(user3);

const element = (
  <div>
    {element1}
    {element2}
    {element3}
  </div>
);

ReactDOM.render(element, document.getElementById("root"));
